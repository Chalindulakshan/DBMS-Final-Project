CREATE TABLE q1_marks (
    m_id VARCHAR(8) NOT NULL,
    Q1 DECIMAL(5,2) NOT NULL,
    status ENUM('proper', 'repeat') NOT NULL DEFAULT 'proper',

    reg_no VARCHAR(8),
